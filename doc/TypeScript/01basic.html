# 接口

# 基本定义

interface Person {
  name: string,
  // 只读属性
  readonly sex: string
  // 只读数组
  family: ReadonlyArray<string>,
  // 可选属性
  age?: number,
  // 当有不确定属性时我们可以如下定义：该接口可以定义任何key为string,值为any类型的属性
  [propName: string]: any
}

let p: Person = { name: 'muyian', sex: 'man', age: 18, family: ['baba', 'sister'] }

// sex和 family是只读属性，无法被修改，下面将会报错
 p.sex = "woman"
 p.family.push("mother")

# 定义方法

interface PrintPerson {
    (name: string, age: number): string
}

let printPerson: PrintPerson = function (name, age) {
    return `name:${name}，age:${age}`
}
let info = printPerson(p.name, p.age)
console.log(info);

# 定义索引类型的接口

interface StringArray {
    [index: number]: string
}
let nameArr: StringArray = ['muyian', 'King'];
console.log(nameArr[0]);

同时typescript也支持字符串索引，两者也可以同时出现，但是数字索引的值必须是字符串索引的值的子类索引类型的接口里也可以添加其他属性，但是其类型必须和索引类型的值类型一致
